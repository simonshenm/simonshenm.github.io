PyTorch实践：迁移学习实例

一. 迁移学习简介

1. 定义

迁移学习是一种将任务A开发的模型作为起点重新使用在任务B中的机器学习。迁移学习就是把预训练好的模型迁移到新的任务上。

2. 2个应用场景

①特征提取

冻结除最终完全连接层之外的所有网络的权重，最后一个全连接层被替换为具有随机权重的新层，并且仅训练该层。

②微调

使用预训练网络初始化网络而不是随机初始化，用新数据训练部分或整个网络。

二. 特征提取

在特征提取中，在预训练网络结构中，修改/添加一个分类器，将源任务上的预训练网络作为另一个目标任务的特征提取器，只对最后增加的分类器参数进行学习，而预先训练好的网络参数不会被修改或冻结。

1. PyTorch提供的预处理模块

①调用随机权重的模型
import torchvision.models as models
resnet18=models.resnet18()
alexnet=models.alexnet()
vgg16=models.vgg16()
②获取预训练模型
import torchvision.models as models
resnet18=models.resnet18(pretrained=True)
alexnet=models.alexnet(pretrained=True)
vgg16=models.vgg16(pretrained=True)
③模式切换
models.train()
models.eval()
⑤规范化数据

所有预训练模型都要求输入图片以相同方式进行标准化，mini-batch的3通道RGB格式(3HW)，其中H，W都应小于224，图像加载时像素值范围应在0到1之间，指定mean=[.485,.456,.406]，std=[.229,.224,.225]

normalize=transforms.Normalize(mean=[.485,.456,.406],std=[.229,.224,.225])

#8. 训练及验证模型
def get_acc(output,label):
    total=output.shape[0]
    _,pred_label=output.max(1)
    num_correct=(pred_label==label).sum().item()
    return num_correct/total 

def train(net,train_data,valid_data,num_epoches,optimizer,criterion):
    prev_time=datetime.now()
    for epoch in range(num_epoches):
        train_loss=0
        train_acc=0
        net=net.train()
        for im,label in train_data:
            im=im.to(device)
            label=label.to(device)
            output=net(im)
            loss=criterion(output,label)
            optimizer.zero_grad()
            loss.backward()
            optimizer.step()
            
            train_loss+=loss.item()
            train_acc+=get_acc(output,label)
        cur_time=datetime.now()
        h,remainder=divmod((cur_time-prev_time).seconds,3600)
        m,s=divmod(remainder,60)
        time_str="Time %02d:%02d:%02d:%02d" % (h,m,s)
        if valid_data is not None:
            valid_loss=0
            valid_acc=0
            net=net.eval()
            for im,label in valid_data:
                im=im.to(device)
                label=label.to(device)
                output=net(im)
                loss=criterion(output,label)
                valid_loss+=loss.item()
                valid_acc+=get_acc(output,label)
            epoch_str=(
                "Epoch %d. Train Loss:%f, Train Acc:%f, Valid Loss:%f, Valid Acc:%f" 
                % (epoch,train_loss/len(train_data),train_acc/len(train_data),valid_loss/len(valid_data),valid_acc/len(valid_data))
            )
        else:
            epoch_str=(
                "Epoch %d. Train Loss:%f, Train Acc:%f"
                % (epoch,train_loss/len(train_data),train_acc/len(train_data))
            )
        prev_time=cur_time
        if epoch%2==0:
            print(epoch_str+time_str)
            
train(net,trainloader,testloader,20,optimizer,criterion)

class model(nn.Module):
    def __init__(self):
        super(model,self).__init__()
        self.relu=nn.ReLU(inplace=True)
        
        self.e_conv1=nn.Conv2d(3,3,1,1,0,bias=True)
        self.e_conv2=nn.Conv2d(3,3,3,1,1,bias=True)
        self.e_conv3=nn.Conv2d(6,3,5,1,2,bias=True)
        self.e_conv4=nn.Conv2d(6,3,7,1,3,bias=True)
        self.e_conv5=nn.Conv2d(12,3,3,1,1,bias=True)
        
    def forward(self,x):
        source=[]
        source.append(x)
        
        x1=self.relu(self.e_conv1(x))
        x2=self.relu(self.e_conv2(x1))
        concat1=torch.cat((x1,x2),1)
        x3=self.relu(self.e_conv3(concat1))
        concat2=torch.cat((x2,x3),1)
        x4=self.relu(self.e_conv4(concat2))
        concat3=torch.cat((x1,x2,x3,x4),1)
        x5=self.relu(self.e_conv5(concat3))
        clean_img=self.relu((x5*x)-x5+1)
        return clean_img
    
#4. 训练模型
device=torch.device("cuda:0" if torch.cuda.is_available() else "cpu")
